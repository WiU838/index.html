"""Оплата бронирования. Реальной интеграции со Сбербанком нет (SBER_MERCHANT_LOGIN/PASSWORD не заданы), поэтому оплата — тестовая заглушка: сразу создаёт успешный Payment и переводит бронирование в статус PAID. Когда появятся боевые реквизиты Сбер Эквайринга, здесь нужно будет создавать платёжную форму вместо мгновенного подтверждения. """ import logging from aiogram import Router, F from aiogram.types import CallbackQuery from sqlalchemy import select, and_ from sqlalchemy.orm import selectinload from core.models import User, Booking, BookingStatus, Payment, PaymentStatus, AsyncSessionLocal logger = logging.getLogger(__name__) router = Router() @router.callback_query(F.data.startswith("pay_booking_")) async def pay_booking(callback: CallbackQuery): booking_id = int(callback.data.removeprefix("pay_booking_")) async with AsyncSessionLocal() as db: user = (await db.execute( select(User).where(User.telegram_id == callback.from_user.id) )).scalar_one_or_none() booking = (await db.execute( select(Booking) .where(Booking.id == booking_id) .options(selectinload(Booking.spot)) )).scalar_one_or_none() if not booking or not user or booking.user_id != user.id: await callback.answer("Бронирование не найдено", show_alert=True) return if booking.status != BookingStatus.PENDING: await callback.answer("Это бронирование уже оплачено или отменено", show_alert=True) return # Место могло уйти к другому платёжеспособному клиенту, пока это # бронирование висело неоплаченным — перепроверяем перед оплатой. conflict = (await db.execute( select(Booking).where( and_( Booking.id != booking.id, Booking.spot_id == booking.spot_id, Booking.status.in_([ BookingStatus.PAID, BookingStatus.CONFIRMED, BookingStatus.ACTIVE, ]), Booking.start_time < booking.end_time, Booking.end_time > booking.start_time, ) ) )).scalar_one_or_none() if conflict: booking.status = BookingStatus.CANCELLED await db.commit() await callback.answer( "К сожалению, место заняли раньше вас. Бронирование отменено.", show_alert=True ) return db.add(Payment( booking_id=booking.id, amount=booking.total_amount, currency="RUB", status=PaymentStatus.SUCCESS, description="Тестовая оплата (SBER_TEST_MODE)", )) booking.status = BookingStatus.PAID await db.commit() spot_number = booking.spot.spot_number if booking.spot else f"ID:{booking.spot_id}" total_amount = booking.total_amount booking_number = booking.id logger.info("Booking %s paid", booking_number) await callback.message.edit_text( f"✅ *Оплата прошла успешно!*\n\n" f"🅿️ Место: *{spot_number}*\n" f"💰 Сумма: *{total_amount}₽*\n" f"Статус: *Оплачено*\n\n" f"Бронирование #{booking_number} подтверждено.", parse_mode="Markdown" ) await callback.answer("Оплата прошла успешно ✅")
